```python
"""Pydantic数据模型定义"""

from pydantic import BaseModel, Field
from typing import List, Optional
from enum import Enum


class HallucinationType(str, Enum):
    """幻觉类型枚举"""
    DIAGNOSIS_ERROR = "diagnosis_error"
    MEDICATION_ERROR = "medication_error"
    EXAM_RESULT_ERROR = "exam_result_error"
    TIME_ERROR = "time_error"
    VALUE_ERROR = "value_error"
    NEGATION_ERROR = "negation_error"
    INVENTED_FACT = "invented_fact"


class EvidenceGrade(str, Enum):
    """证据等级枚举"""
    E1 = "E1"  # 结构化字段支持
    E2 = "E2"  # 非结构化证据支持
    E3 = "E3"  # 无证据支持
    E4 = "E4"  # 与已有证据冲突


class HallucinationDetectionResult(BaseModel):
    """幻觉检测结果 - AI输出部分"""
    reasoning: str = Field(..., description="有无幻觉的原因和思考过程")
    hallucination_status: bool = Field(..., description="是否存在幻觉,True表示存在")
    hallucination_type: Optional[HallucinationType] = Field(None, description="幻觉类型,仅当存在幻觉时填写")
    evidence_grade: EvidenceGrade = Field(..., description="证据等级")


class SentenceDetectionRecord(BaseModel):
    """单个句子的完整检测记录"""
    sentence_index: int = Field(..., description="句子编号")
    original_sentence: str = Field(..., description="原始句子")
    detection_result: HallucinationDetectionResult = Field(..., description="AI检测结果")
    graphrag_evidence: str = Field(..., description="GraphRAG提供的原始证据")


class DetectionReport(BaseModel):
    """完整的检测报告"""
    patient_id: str = Field(..., description="患者ID")
    total_sentences: int = Field(..., description="总句子数")
    detected_hallucinations: int = Field(..., description="检测到的幻觉数量")
    hallucination_rate: float = Field(..., description="幻觉比例")
    records: List[SentenceDetectionRecord] = Field(..., description="所有检测记录")


class GraphRAGContext(BaseModel):
    """GraphRAG返回的上下文数据"""
    entities: List[str] = Field(default_factory=list, description="相关实体列表")
    relationships: List[str] = Field(default_factory=list, description="相关关系列表")
    community_reports: List[str] = Field(default_factory=list, description="社区报告列表")
    raw_response: str = Field(..., description="GraphRAG的原始响应文本")
```

```python
def setup_logger(name: str = "graphrag_detection", log_dir: str = "./logs/agent_logs") -> logging.Logger:
```

```python
"""句子分割工具"""

import re
from typing import List, Tuple
from langchain.tools import tool

@tool
def split_sentences(text: str) -> List[Tuple[int, str]]:
    """
    将文本分割为句子列表,返回(索引, 句子)元组
    
    Args:
        text: 原始文本
    
    Returns:
        [(0, "句子1"), (1, "句子2"), ...]
    """
    # 正则表达式: 处理中文句号、英文句号、问号、感叹号
    # 注意保留缩写 (如 Dr., Mr.)
    sentence_endings = re.compile(r'([。!?;；\n]|(?<!\d)\.(?!\d))')
    
    # 分割
    raw_sentences = sentence_endings.split(text)
    
    # 清洗和重组
    sentences = []
    current = ""
    for part in raw_sentences:
        if part in ['。', '.', '!', '?', ';', '；']:
            if current.strip():
                sentences.append(current.strip() + part)
            current = ""
        else:
            current += part
    
    # 添加最后一个句子
    if current.strip():
        sentences.append(current.strip())
    
    # 过滤空句子和过短句子
    filtered = [(i, s) for i, s in enumerate(sentences) if len(s) >= 10]
    
    return filtered
```

```python
async def _async_query_graphrag(
    sentence: str,
    patient_id: str,
    method: str = "local"
) -> dict:
    """
    异步查询GraphRAG (内部函数)
    
    Returns:
        {
            "response": "GraphRAG的文本回答",
            "context": {
                "entities": [...],
                "relationships": [...],
                "reports": [...]
            }
        }
    """
    project_root = Path("graphrag")
    index_root = project_root / "output" / "index_output" / patient_id
    
    if not index_root.exists():
        raise ValueError(f"患者索引不存在: {patient_id}")
    
    # 加载配置和数据
    config = load_config(index_root)
    output_dir = index_root / "output"
    
    entities = pd.read_parquet(output_dir / "entities.parquet")
    relationships = pd.read_parquet(output_dir / "relationships.parquet")
    text_units = pd.read_parquet(output_dir / "text_units.parquet")
    communities = pd.read_parquet(output_dir / "communities.parquet")
    community_reports = pd.read_parquet(output_dir / "community_reports.parquet")
    
    # 执行查询
    if method == "local":
        response, context_data = await api.local_search(
            config=config,
            entities=entities,
            communities=communities,
            community_reports=community_reports,
            text_units=text_units,
            relationships=relationships,
            covariates=None,
            community_level=2,
            response_type="Multiple Paragraphs",
            query=f"请提取与以下句子相关的所有医学信息: {sentence}",
        )
    else:
        raise ValueError(f"不支持的查询方法: {method}")
    
    # 提取关键信息
    context = {
        "entities": _extract_entities(context_data),
        "relationships": _extract_relationships(context_data),
        "reports": _extract_reports(context_data),
    }
    
    return {
        "response": response,
        "context": context
    }


def _extract_entities(context_data) -> list:
    """从context_data提取实体信息"""
    # 根据GraphRAG的实际数据结构提取
    # 这里需要根据实际运行结果调整
    if hasattr(context_data, 'entities'):
        return [str(e) for e in context_data.entities[:10]]
    return []


def _extract_relationships(context_data) -> list:
    """从context_data提取关系信息"""
    if hasattr(context_data, 'relationships'):
        return [str(r) for r in context_data.relationships[:10]]
    return []


def _extract_reports(context_data) -> list:
    """从context_data提取报告摘要"""
    if hasattr(context_data, 'reports'):
        return [str(r)[:200] for r in context_data.reports[:3]]
    return []
```

#### Step 3.4.3 实现同步Tool接口
```python
@tool
def query_graphrag_for_sentence(sentence: str, patient_id: str = "patient_2") -> str:
    """
    使用GraphRAG查询与句子相关的医学知识
    
    Args:
        sentence: 待检测的句子
        patient_id: 患者ID
    
    Returns:
        JSON格式的查询结果
    """
    try:
        # 创建新的事件循环 (解决asyncio冲突)
        loop = asyncio.new_event_loop()
        asyncio.set_event_loop(loop)
        result = loop.run_until_complete(_async_query_graphrag(sentence, patient_id))
        loop.close()
        
        # 序列化为JSON
        return json.dumps(result, ensure_ascii=False, indent=2)
    
    except Exception as e:
        logger.error(f"GraphRAG查询失败: {e}")
        return json.dumps({
            "error": str(e),
            "response": "查询失败",
            "context": {"entities": [], "relationships": [], "reports": []}
        })
```

### Step 3.5 实现 `prompts/detection_prompt.md`
```markdown
# 系统角色定义
你是一名专业的医学幻觉检测专家。你的任务是检测医学文本中的幻觉(与事实不符的陈述)。

## 幻觉类型定义
请判断以下句子是否属于这些幻觉类型:

1. **diagnosis_error** (诊断错误): 诊断名称错误或与证据不符
   示例: 实际诊断是"急性心肌梗死",却写成"心绞痛"

2. **medication_error** (用药错误): 药物名称、剂量、用法错误
   示例: 实际使用"阿司匹林100mg",却写成"阿托伐他汀100mg"

3. **exam_result_error** (检查结果错误): 检验检查结果数值或描述错误
   示例: 实际心电图正常,却写成"ST段抬高"

4. **time_error** (时间错误): 入院时间、手术时间等时间信息错误
   示例: 实际入院时间2023-01-05,却写成2023-01-01

5. **value_error** (数值错误): 体征、化验等数值错误
   示例: 实际血压140/90,却写成120/80

6. **negation_error** (否定错误): 肯定/否定描述相反
   示例: 实际"无发热",却写成"有发热"

7. **invented_fact** (无中生有): 完全编造的、证据中不存在的事实
   示例: 凭空添加"患者有糖尿病史10年"

## 证据等级定义
请根据GraphRAG提供的证据,判断证据等级:

- **E1** (结构化字段支持): 在GraphRAG的实体或关系中有明确的结构化数据支持
  判断标准: entities或relationships字段中有直接匹配的信息

- **E2** (非结构化证据支持): 在文本描述或报告中有支持
  判断标准: reports字段中有相关描述

- **E3** (无证据支持): GraphRAG返回的证据中既无结构化也无非结构化支持
  判断标准: 所有字段都找不到相关信息

- **E4** (与已有证据冲突): 与GraphRAG提供的证据直接矛盾
  判断标准: 证据中的信息与句子陈述相反

## 检测流程
请按照以下步骤进行检测:

1. **理解句子**: 提取句子中的关键医学事实(诊断、用药、数值等)
2. **查看证据**: 仔细阅读GraphRAG提供的entities、relationships、reports
3. **逐项比对**: 将句子中的每个事实与证据比对
4. **判断类型**: 如果存在幻觉,确定属于哪种类型
5. **评估等级**: 根据证据的来源和强度,确定证据等级

## 输出要求
请输出你的检测结果,格式如下:

- **reasoning**: 详细说明你的推理过程,包括:
  - 句子中包含哪些关键事实
  - GraphRAG证据中有哪些相关信息
  - 是否存在矛盾或不一致
  - 最终判断的依据

- **hallucination_status**: true(存在幻觉) 或 false(无幻觉)

- **hallucination_type**: 如果存在幻觉,选择一个最匹配的类型;如果无幻觉,填null

- **evidence_grade**: E1/E2/E3/E4 四选一

## 注意事项
- 如果证据不足,宁可标记为E3,也不要随意判断为幻觉
- 医学术语的同义表达不算幻觉 (如"心梗"="心肌梗死")
- 数值的轻微差异如果在合理范围内不算幻觉 (如血压测量误差±5mmHg)
- 必须基于证据,不要凭主观臆断
```

```python
"""GraphRAG幻觉检测Agent定义"""

import os
import yaml
from langchain.agents import create_agent
from langchain_openai import ChatOpenAI
from langgraph.checkpoint.memory import InMemorySaver

from tools.sentence_splitter import split_sentences
from tools.graphrag_query_tool import query_graphrag_for_sentence
from models.schemas import HallucinationDetectionResult
from utils.prompt_loader import load_detection_prompt
from utils.logger import setup_logger

logger = setup_logger()

# 加载配置
config_path = os.path.join(os.path.dirname(__file__), 'config.yaml')
with open(config_path, 'r', encoding='utf-8') as f:
    CONFIG = yaml.safe_load(f)


def create_detection_agent():
    """
    创建幻觉检测Agent
    
    Returns:
        配置好的Agent实例
    """
    # 初始化LLM
    llm_config = CONFIG['llm']
    llm = ChatOpenAI(
        model=llm_config['model'],
        temperature=llm_config['temperature'],
        max_tokens=llm_config['max_tokens'],
        api_key=os.getenv(llm_config['api_key'].strip('${}')),
        base_url=llm_config.get('base_url')
    )
    
    # 配置结构化输出
    llm_structured = llm.with_structured_output(HallucinationDetectionResult)
    
    # 加载系统提示词
    system_prompt = load_detection_prompt()
    
    # 定义工具列表
    tools = [
        split_sentences,
        query_graphrag_for_sentence
    ]
    
    # 创建Agent
    agent = create_agent(
        model=llm,
        tools=tools,
        system_prompt=system_prompt,
        checkpointer=InMemorySaver()
    )
    
    logger.info("Agent创建成功")
    return agent, llm_structured


def detect_single_sentence(
    sentence: str,
    patient_id: str,
    llm_structured
) -> dict:
    """
    检测单个句子的幻觉
    
    Args:
        sentence: 待检测的句子
        patient_id: 患者ID
        llm_structured: 配置了结构化输出的LLM
    
    Returns:
        检测结果字典
    """
    # Step 1: 调用GraphRAG查询
    logger.info(f"正在查询GraphRAG: {sentence[:50]}...")
    graphrag_result = query_graphrag_for_sentence.invoke({
        "sentence": sentence,
        "patient_id": patient_id
    })
    
    # Step 2: 构造检测提示词
    user_prompt = f"""
待检测句子: "{sentence}"

GraphRAG检索到的相关证据:
{graphrag_result}

请基于以上证据,检测该句子是否存在医学幻觉。
"""
    
    # Step 3: 调用LLM进行检测
    logger.info("正在进行幻觉检测...")
    detection_result = llm_structured.invoke([
        {"role": "system", "content": load_detection_prompt()},
        {"role": "user", "content": user_prompt}
    ])
    
    # Step 4: 组装结果
    return {
        "detection_result": detection_result,
        "graphrag_evidence": graphrag_result
    }
```

```python
def process_text_file(
    text_file_path: str,
    patient_id: str = None

    """

    logger.info("=" * 60)

    logger.info("=" * 60)
    
    # Step 1: 读取文件
    logger.info(f"读取文件: {text_file_path}")

        full_text = f.read()

    sentence_list = split_sentences.invoke({"text": full_text})
    total_sentences = len(sentence_list)
    logger.info(f"总句子数: {total_sentences}")
    
    # Step 3: 创建Agent
    logger.info("初始化Agent...")
    agent, llm_structured = create_detection_agent()
    
    # Step 4: 逐句检测
    records: List[SentenceDetectionRecord] = []
    hallucination_count = 0
    
    for idx, sentence in sentence_list:
        logger.info(f"\n[{idx+1}/{total_sentences}] 检测句子: {sentence[:60]}...")
        
        try:
            result = detect_single_sentence(sentence, patient_id, llm_structured)
            
            record = SentenceDetectionRecord(
                sentence_index=idx,
                original_sentence=sentence,
                detection_result=result['detection_result'],
                graphrag_evidence=result['graphrag_evidence']
            )
            
            records.append(record)
            
            if result['detection_result'].hallucination_status:
                hallucination_count += 1
                logger.warning(f"  ⚠️  检测到幻觉: {result['detection_result'].hallucination_type}")
            else:
                logger.info(f"  ✓ 无幻觉")
        
        except Exception as e:
            logger.error(f"  ❌ 检测失败: {e}")
            continue
    
    # Step 5: 生成报告
    report = DetectionReport(
        patient_id=patient_id,
        total_sentences=total_sentences,
        detected_hallucinations=hallucination_count,
        hallucination_rate=hallucination_count / total_sentences if total_sentences > 0 else 0,
        records=records
    )
    
    # Step 6: 保存结果
    if CONFIG['output']['save_to_file']:
        output_dir = CONFIG['output']['output_dir']
        os.makedirs(output_dir, exist_ok=True)
        
        output_file = os.path.join(output_dir, f"{patient_id}_detection_report.json")
        with open(output_file, 'w', encoding='utf-8') as f:
            f.write(report.model_dump_json(indent=2, ensure_ascii=False))
        
        logger.info(f"\n✅ 报告已保存: {output_file}")
    
    # 打印摘要
    logger.info("\n" + "=" * 60)
    logger.info("检测摘要")
    logger.info("=" * 60)
    logger.info(f"总句子数: {total_sentences}")
    logger.info(f"检测到幻觉: {hallucination_count}")
    logger.info(f"幻觉比例: {report.hallucination_rate:.2%}")
    logger.info("=" * 60)
    
    return report


if __name__ == "__main__":
    import argparse
    
    parser = argparse.ArgumentParser(description='GraphRAG幻觉检测Agent')
    parser.add_argument('text_file', help='待检测的文本文件路径')
    parser.add_argument('--patient-id', default=None, help='患者ID')
    
    args = parser.parse_args()
    
    report = process_text_file(args.text_file, args.patient_id)
```
